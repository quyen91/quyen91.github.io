### Inject with ruby array
Dùng inject có thể làm cho code được gọn hơn
```ruby
 # With each
 def sum
   result = 0
    [1, 2, 3, 4].each do |e|
      result += e
    end
    result # => 10
  end

  # With inject
  def sum
    [1, 2, 3, 4].inject(0) { |result, element| result + element } # => 10
  end
```
